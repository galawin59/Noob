name= nom de l'objet (string)
type= type de l'objet (convertit en TypeItem)
pref= racine du nom de préfab (string)
nbrD= directions possibles pour l'objet (int à 4 chiffres => [top bot right left] avec 0 = false / 1 = true)
nbrP= nombre de parties de l'objet (int)
snap= si la position de l'objet dépends du grid (bool)
inte= si l'objet possède des interactions (bool)
cbpo= le nom des objets sur lesquels cet objet peut etre posé (string)
nbrS= le nombres de sprites differents pour cette item (int)
